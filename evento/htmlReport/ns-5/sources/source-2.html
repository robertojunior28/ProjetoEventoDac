 package br.edu.ifpb.dac.junior.model.entity;
 
 import jakarta.persistence.*;
 import lombok.*;
 
 @Getter
 @Setter
 @NoArgsConstructor
 @AllArgsConstructor
 @ToString
 @Entity
 @Table(name = "local")
 public class Local {
     @Id
     @GeneratedValue(strategy = GenerationType.AUTO)
     private Long id;
     private String street;
     private String number;
     private String city;
     private String uf;
 
     @ManyToOne
     @JoinColumn(name = "event_id")
     private Event event;
 
     public Local(String street, String number, String city, String uf) {
         this.street = street;
         this.number = number;
         this.city = city;
         this.uf = uf;
     }
 
 
 }
